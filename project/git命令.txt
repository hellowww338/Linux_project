git pull origin main （--allow-unrelated-histories）  获取最新更改
git add .		添加暂存区
git commit -m "Your commit message"			提交信息
git push origin main

更新git data 到本地：
git fetch origin			查看更新，不合并
git pull origin main		//合并更新


git remote -v       # 查看远程仓库配置
git remote remove origin删除
